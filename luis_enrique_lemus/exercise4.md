class Operaciones{
  double? num1;
  double? num2;
  
  double? sumar(){
    double s = num1! + num2!;
    return s;
  }
  
  void resta(){
    double r = num1! - num2!;
    print("El resultado es; $r");
  }
  
  double? multiplicar(){
    double m = num1! * num2!;
    return m;
  }
}
