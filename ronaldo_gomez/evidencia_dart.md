class Operaciones {
  double? num1;
  double? num2;
  
  double sumar() {
    double sum = num1! + num2!;
    return sum;
  }
  double multiplicar() {
    double mul = num1! * num2!;
    return mul;
  }
  void restar() {
    double res = num1! - num2!;
    print('El total de la resta es: $res');
  }
}
